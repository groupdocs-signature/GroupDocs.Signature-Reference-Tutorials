# วิธีการลงนามและแปลงงานนำเสนอด้วย GroupDocs.Signature สำหรับ .NET: คู่มือที่ครอบคลุม

## การแนะนำ

ในยุคดิจิทัล การรักษาความปลอดภัยเอกสารเป็นสิ่งสำคัญอย่างยิ่ง โดยเฉพาะงานนำเสนอที่มักมีข้อมูลสำคัญ ด้วย GroupDocs.Signature สำหรับ .NET คุณสามารถลงนามในงานนำเสนอและแปลงเป็นรูปแบบอื่นได้อย่างง่ายดายโดยใช้โค้ดเพียงไม่กี่บรรทัด บทช่วยสอนนี้จะแนะนำคุณตลอดการผสานรวมลายเซ็นดิจิทัลและการแปลงไฟล์อย่างราบรื่น เพื่อให้มั่นใจว่าเอกสารของคุณมีความปลอดภัยและใช้งานได้หลากหลาย

## คู่มือการใช้งาน

### การลงนามในงานนำเสนอและการบันทึกเป็นไฟล์ประเภทอื่น

เพิ่มลายเซ็นดิจิทัลลงในงานนำเสนอและบันทึกในรูปแบบต่างๆ:

#### สร้างตัวเลือก QRCodeSignOptions
กำหนดคุณสมบัติของลายเซ็น QR code ของคุณโดยใช้ `QrCodeSignOptions`-

```csharp
using GroupDocs.Signature.Options;

// กำหนดตัวเลือกลายเซ็นรหัส QR
QrCodeSignOptions signOptions = new QrCodeSignOptions("JohnSmith")
{
    EncodeType = QrCodeTypes.QR,
    Left = 100, // ตำแหน่งแนวนอนบนหน้า
    Top = 100   // ตำแหน่งแนวตั้งบนหน้า
};
```

#### ตั้งค่าตัวเลือกการบันทึกการนำเสนอ
ระบุวิธีที่คุณต้องการบันทึกเอกสารที่ลงนามของคุณโดยใช้ `PresentationSaveOptions`-

```csharp
using GroupDocs.Signature.Domain;

// กำหนดค่าตัวเลือกการบันทึกสำหรับการนำเสนอที่ลงนาม
PresentationSaveOptions saveOptions = new PresentationSaveOptions()
{
    FileFormat = PresentationSaveFileFormat.Tiff,
    OverwriteExistingFiles = true
};
```

#### ลงชื่อและบันทึก
ลงนามในเอกสารของคุณและบันทึกในรูปแบบที่ต้องการ:

```csharp
using GroupDocs.Signature;

// ดำเนินการลงนามและบันทึกกระบวนการ
SignResult result = signature.Sign("output/path", signOptions, saveOptions);
```

### การตั้งค่าเส้นทางเอกสาร
ตั้งค่าเส้นทางสำหรับไฟล์อินพุตและเอาต์พุตอย่างถูกต้อง:

```csharp
string sourceDocumentPath = Path.Combine("YOUR_DOCUMENT_DIRECTORY", "Sample_Document.docx");
string signedOutputPath = Path.Combine("YOUR_OUTPUT_DIRECTORY", "SignedDocuments", "Signed_Document.pdf");
```
